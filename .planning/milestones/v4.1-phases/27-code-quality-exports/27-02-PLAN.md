    ```typescript
    async function exportJson(): Promise<void> {
      if (isExporting.value) return; // prevent double-click (POLISH-03 pattern, per D-10)
      const userId = pb.authStore.record?.id;
      if (!userId) {
        toast.error("Session expired. Please log in again."); // per D-11
        return;
      }
      isExporting.value = true;
      try {
        const allRecords = await pb
          .collection("wallecx_memberships")
          .getFullList<Memberships>({
            sort: "-created",           // per D-12
            requestKey: "memberships-export", // per D-18 — distinct from 'memberships-getFullList' to avoid cancelling the live data subscription
          });

        const exportPayload = {
          exported_at: new Date().toISOString(), // per D-13
          record_count: allRecords.length,
          records: allRecords.map((r) => ({     // per D-04 — writable domain fields only
            id: r.id,
            card_name: r.card_name,
            issuer: r.issuer ?? null,
            card_number: r.card_number ?? null,
            card_color: r.card_color ?? null,
            expiry_date: r.expiry_date ?? null,
            notes: r.notes ?? null,
            card_image_url: r.card_image ? pb.files.getURL(r, r.card_image) : null, // per D-03
            created: r.created,
            updated: r.updated,
            // NOT included: user, collectionId, collectionName, barcode_value, barcode_type
          })),
        };

        // per D-14 — Blob download + cleanup
        const blob = new Blob([JSON.stringify(exportPayload, null, 2)], {
          type: "application/json",
        });
        const url = URL.createObjectURL(blob);
        const anchor = document.createElement("a");
        anchor.href = url;
        anchor.download = `wallecx-memberships-${dayjs().format("YYYY-MM-DD")}.json`; // per D-15
        document.body.appendChild(anchor);
        anchor.click();
        document.body.removeChild(anchor);
        URL.revokeObjectURL(url);

        toast.success("Membership cards exported."); // per D-16
      } catch (e: unknown) {
        toast.error("Export failed. Please try again."); // per D-16
        console.error("MembershipsTab: exportJson failed", e);
      } finally {
        isExporting.value = false; // MANDATORY — prevents stuck loading state on error
      }
    }
    ```
