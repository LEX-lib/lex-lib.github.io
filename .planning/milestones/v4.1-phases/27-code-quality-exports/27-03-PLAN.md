    ```typescript
    async function exportJson(): Promise<void> {
      if (isExporting.value) return; // prevent double-click (POLISH-03 pattern, per D-10)
      const userId = pb.authStore.record?.id;
      if (!userId) {
        toast.error("Session expired. Please log in again."); // per D-11
        return;
      }
      isExporting.value = true;
      try {
        const allRecords = await pb
          .collection("wallecx_expenses")
          .getFullList<Expenses>({
            sort: "-expense_date,-created",   // per D-12
            requestKey: "expenses-export",    // per D-18 — distinct from 'expenses-getFullList' to avoid cancelling the live data subscription
          });

        const exportPayload = {
          exported_at: new Date().toISOString(), // per D-13
          record_count: allRecords.length,
          records: allRecords.map((r) => ({     // per D-07 — writable domain fields only
            id: r.id,
            amount: r.amount,
            expense_date: r.expense_date,
            category: r.category,
            description: r.description,
            notes: r.notes ?? null,
            receipt_url: r.receipt ? pb.files.getURL(r, r.receipt) : null, // per D-06
            created: r.created,
            updated: r.updated,
            // NOT included: user, collectionId, collectionName, raw receipt filename
          })),
        };

        // per D-14 — Blob download + cleanup
        const blob = new Blob([JSON.stringify(exportPayload, null, 2)], {
          type: "application/json",
        });
        const url = URL.createObjectURL(blob);
        const anchor = document.createElement("a");
        anchor.href = url;
        anchor.download = `wallecx-expenses-${dayjs().format("YYYY-MM-DD")}.json`; // per D-15
        document.body.appendChild(anchor);
        anchor.click();
        document.body.removeChild(anchor);
        URL.revokeObjectURL(url);

        toast.success("Expenses exported."); // per D-16
      } catch (e: unknown) {
        toast.error("Export failed. Please try again."); // per D-16
        console.error("ExpensesTab: exportJson failed", e);
      } finally {
        isExporting.value = false; // MANDATORY — prevents stuck loading state on error
      }
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → PocketBase API | `getFullList` call fetches all user expense records — requires valid auth token |
| Browser Blob API | Exported JSON is written to client memory and downloaded via anchor click — no server involvement |
| User session | `pb.authStore.record?.id` null guard prevents unauthenticated export |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-27-03-01 | Information Disclosure | exportJson() — unauthenticated call | mitigate | `pb.authStore.record?.id` null guard (D-11) aborts export and shows "Session expired" toast before any PocketBase call is made |
| T-27-03-02 | Information Disclosure | PocketBase getFullList returns cross-user records | accept | PocketBase listRule enforces `@request.auth.id != "" && @request.auth.id = user` server-side (locked architectural invariant). Client guard is belt-and-suspenders. |
| T-27-03-03 | Information Disclosure | `receipt_url` in exported JSON — protected=true field | accept | Documented in D-06 / RESEARCH.md Security Domain. Receipt URLs require a short-lived file token to access (protected=true); raw URL without token returns 403. Included in export for backup reference only; acceptable per locked decision. This is the documented behaviour from Phase 23 (receipt File field: protected=true). |
| T-27-03-04 | Denial of Service | Double-click triggering concurrent getFullList calls | mitigate | `isExporting` ref guard (D-10) returns early on second click. Button is also `disabled` while `isExporting` is true, providing UI-level protection. |
| T-27-03-05 | Tampering | Blob URL lingering in memory | mitigate | `URL.revokeObjectURL(url)` called immediately after `anchor.click()` per D-14 pattern. Object URL is revoked before the function returns. |
| T-27-03-06 | Information Disclosure | Export bypasses active UI filters | accept | This is intended behaviour (D-08). The export calls `getFullList` fresh — not the filtered `filteredSortedExpenses` computed. Users export ALL their records regardless of currently active date range or category filter.
