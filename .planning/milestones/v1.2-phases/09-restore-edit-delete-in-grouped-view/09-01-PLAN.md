  confirm.require({
    message: `Delete "${record.vaccine_name}"? This cannot be undone.`,
    header: "Confirm Delete",
    icon: "pi pi-exclamation-triangle",
    rejectProps: { label: "Keep Record", severity: "secondary", outlined: true },

    accept: () => deleteRecord(record),
