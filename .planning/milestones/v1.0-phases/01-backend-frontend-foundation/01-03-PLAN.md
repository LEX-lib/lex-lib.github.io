    ```vue
    <script setup lang="ts">
    import { ref, onMounted } from "vue";
    import { toast } from "vue-sonner";
    import { pb } from "@/lib/pocketbase";
    import type { Vaccinations } from "@/types/wallecx/vaccinations/types";

    // --- STATE ---
    const records = ref<Vaccinations[]>([]);
    const isLoading = ref(false);

    // --- LOGIC ---
    onMounted(async () => {
      isLoading.value = true;
      try {
        records.value = await pb
          .collection("wallecx_vaccinations")
          .getFullList<Vaccinations>({ sort: "-date_administered" });
      } catch (e: unknown) {
        toast.error("Failed to load vaccination records.");
        console.error("WallecxApp: getFullList failed", e);
      } finally {
        isLoading.value = false;
      }
    });
    </script>

    <template>
      <Card>
        <template #content>
          <div class="flex items-center justify-between mb-4">
            <h1 class="text-2xl font-bold" style="color: var(--color-typo-heading)">
              Wallecx
            </h1>
          </div>
          <div>
            <p v-if="isLoading">Loading...</p>
            <p v-else>{{ records.length }} vaccination record{{ records.length === 1 ? "" : "s" }}</p>
          </div>
        </template>
      </Card>
    </template>

    <style scoped></style>
    ```

    Key implementation details:
    - `<Card>` is auto-imported via PrimeVue resolver — no explicit import needed
    - Page title is `"Wallecx"` per D-05 (brand name, not "Vaccination Records")
    - `sort: "-date_administered"` (descending — newest first)
    - `e: unknown` is used (stricter than `: any`; acceptable for new code per CONVENTIONS.md)
    - Section comments `// --- STATE ---` and `// --- LOGIC ---` follow CONVENTIONS.md pattern

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Vue Router guard → /projects/wallecx | requiresAuth: true activates the beforeEach guard; unauthenticated access redirects to /login |
| WallecxApp.vue → PocketBase API | getFullList is filtered server-side by collection rules; client cannot override them |
| vaccinationMapper.ts return type → pb.update() | Inline return type enforces field stripping at compile time; user and card cannot appear in the payload |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-12 | Elevation of Privilege | /projects/wallecx without auth | mitigate | meta: { requiresAuth: true } activates beforeEach guard; redirects to /login?redirect=/projects/wallecx for unauthenticated sessions |
| T-01-13 | Information Disclosure | vaccinationMapper including `user` field in update payload | mitigate | Inline return type explicitly excludes `user:`; TypeScript compiler rejects any code that tries to include it |
| T-01-14 | Tampering | vaccinationMapper including `card` field in update payload | mitigate | `card` is absent from the inline return type; sending it in an update could overwrite an existing attachment with an empty value |
| T-01-15 | Information Disclosure | pdfjs-dist CVE-2024-4367 (arbitrary JS via malicious PDF) | accept | vue-pdf-embed@2.1.4 depends on pdfjs-dist@^4.10.38 which resolves to 5.x+; CVE was fixed at 4.2.67; no manual pinning needed; verified via package-lock.json |
| T-01-16 | Information Disclosure | card typed as string[] instead of string | accept | Typed as `card: string` per RESEARCH.md Pitfall B; MaxSelect=1 returns plain string; type system prevents array access patterns |
</threat_model>
