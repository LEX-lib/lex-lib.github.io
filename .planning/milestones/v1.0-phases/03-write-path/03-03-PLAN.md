    try {
      const img = new Image();
      const objectUrl = URL.createObjectURL(file);
      await new Promise<void>((resolve, reject) => {
        img.onload = () => resolve();
        img.onerror = reject;
        img.src = objectUrl;
      });

      const canvas = document.createElement("canvas");
      canvas.width = img.naturalWidth;
      canvas.height = img.naturalHeight;
      const ctx = canvas.getContext("2d")!;

      // canvas.toBlob is callback-based — must wrap in Promise (Pitfall D)
      const strippedBlob = await new Promise<Blob>((resolve, reject) =>
        canvas.toBlob(
          (blob) => (blob ? resolve(blob) : reject(new Error("canvas.toBlob failed"))),
          "image/jpeg",
          0.92,
        ),
      );

      // Convert Blob → File so FormData has a filename
      const strippedFile = new File(
        [strippedBlob],
        file.name.replace(/\.[^.]+$/, ".jpg"),
        { type: "image/jpeg" },
      );

      const compressed = await imageCompression(strippedFile, {
        maxSizeMB: 1.5,
        maxWidthOrHeight: 2048,
        useWebWorker: true,
      });

      pendingFile.value = compressed;
      toast.info("Location data removed."); // D-07: unconditional on every image upload

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → PocketBase API | FormData crosses the network; server collection rules are the authoritative access gate |
| Image file → canvas | Untrusted file content processed in browser canvas; canvas re-encode is the EXIF strip |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-01 | Information Disclosure | Image EXIF GPS in uploaded file | mitigate | Canvas re-encode (drawImage → toBlob) strips all EXIF before imageCompression; toast confirms removal (D-07, WRITE-03, Pitfall 7) |
| T-03-03-02 | Tampering | FormData — user field in update | mitigate | "user" is NOT appended in the update branch; PocketBase updateRule checks @request.auth.id == user; sending wrong user would cause 403 |
| T-03-03-03 | Tampering / Integrity | Save-loop duplicate create | mitigate | emit("created", serverRecord) → WallecxApp.onCreated calls Object.assign(localItem, serverRecord) so local item gains server id; subsequent saves correctly PATCH (WRITE-04, Pitfall 3) |
| T-03-03-04 | Tampering | Update PATCH poisoning (server fields) | mitigate | FormData only appends writable fields (vaccine_name, date_administered, optional fields, card); id, created, updated, user are never appended in update branch (WRITE-05) |
| T-03-03-05 | Denial of Service | Large image processing | accept | imageCompression maxSizeMB=1.5 caps output; canvas re-encode may be slow for very large images but this is a personal app; no server-side processing cost |
</threat_model>
