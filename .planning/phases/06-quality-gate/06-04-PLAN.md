const baseData: ExportData = {

Note: `new Date('2026-01-15')` in jsdom will parse as 2026-01-15T00:00:00Z (UTC midnight). Use `dayjs(new Date('2026-01-15')).format('MM/DD/YYYY')` to derive the expected date header in test assertions — OR assert `startsWith('[01/15/2026]')` (the format is fixed, but timezone-safe approach is to call the same dayjs format in tests).
