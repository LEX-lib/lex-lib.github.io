const stripHtml = (html: string): string[] => {

  const doc = new DOMParser().parseFromString(html, 'text/html');
  return Array.from(doc.body.querySelectorAll('p, li, div'))
    .map(el => el.textContent?.trim() ?? '')
    .filter(line => line.length > 0);
