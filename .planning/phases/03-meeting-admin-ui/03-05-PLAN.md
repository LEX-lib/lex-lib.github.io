**`<script setup>`** — wire the composable, watch the v-model meeting prop to re-seed when the
dialog opens with a different meeting, and watch the composable outputs to write back into the
meeting model:

```vue
<script setup lang="ts">
import { watch } from "vue";
import { toast } from "vue-sonner";
import type { AddDsuMeeting } from "@/types/lextrack/dsu_meetings/types";
import {
  DSU_MEETING_DURATION_UNIT_VALUES,
  DSU_MEETING_DURATION_UNIT_LABELS,
  type DurationUnit,
} from "@/types/lextrack/dsu_meetings/constants";
import { useDurationField } from "@/composables/lextrack/useDurationField";

const visible = defineModel(
    'visible',
    {
      type: Boolean,
      default: false,
      required: true,
    });

const meeting = defineModel<AddDsuMeeting>(
    'meeting',
    {
      required: true,
    });

// Seed the composable from the current meeting (D-04 round-trip).
const { enteredValue, unit, durationMinutes, fractionDigits } = useDurationField(
  meeting.value.duration_minutes,
  meeting.value.duration_unit,
);

// Re-seed when the parent swaps which meeting is bound (e.g. user opens a different row).
// Track the meeting reference identity, not deep changes — a deep watcher would fight Edit's writes.
watch(
  () => meeting.value,
  (next) => {
    enteredValue.value =
      next.duration_minutes === undefined || next.duration_minutes === null
        ? null
        : (next.duration_unit ?? 'minutes') === 'hours'
          ? next.duration_minutes / 60
          : next.duration_minutes;
    unit.value = next.duration_unit ?? 'minutes';
  },
);

// Mirror UI changes back into the v-model'd meeting (D-03).
watch([durationMinutes, unit], ([nextMinutes, nextUnit]) => {
  meeting.value.duration_minutes = nextMinutes;
  meeting.value.duration_unit = nextUnit;
});

// Options for the SelectButton (D-06).
const durationUnitOptions = DSU_MEETING_DURATION_UNIT_VALUES.map((value) => ({
  value,
  label: DSU_MEETING_DURATION_UNIT_LABELS[value],
}));

const updateMeeting = () => {
  // Phase 3 boundary: persistence lands in Phase 4 (UI-SAVE-01). For now, toast only.
  toast.success('Meeting is updated successfully!');
};
</script>
```

Notes for the executor:
- The original file imported `ref` (unused) and `Toaster` from vue-sonner (unused). Drop both.
- The original `editorStyle` constant goes away with the `:pt` strip (D-18).
- DO NOT add `import SelectButton from 'primevue/selectbutton'` or any other PrimeVue component
  import. SelectButton, InputNumber, InputText, Dialog, Editor, Button are all auto-imported via
  `unplugin-vue-components` per CLAUDE.md.
- The `DurationUnit` import is `import type { ... }` — used for typing only. The
  `DSU_MEETING_DURATION_UNIT_VALUES` and `DSU_MEETING_DURATION_UNIT_LABELS` are runtime values
  imported normally.
- Watching `() => meeting.value` (reference identity) avoids the watcher firing on every
  `meeting.value.duration_minutes = X` write the second watcher does — preventing a feedback loop.
  Vue's `defineModel` returns a ref whose `.value` is replaced by the parent when binding swaps,
  so reference-identity watching is the correct strategy here.
