Create `src/composables/useTheme.ts` with EXACTLY this body, copied verbatim from CONTEXT.md §Specific Ideas (the snippet implements D-03 through D-09 and D-15):

```ts
import { ref, readonly, type Ref } from 'vue'

const STORAGE_KEY = 'lexarium:theme'
const DARK_CLASS = 'my-app-dark'

type Theme = 'light' | 'dark'

const theme = ref<Theme>(
  document.documentElement.classList.contains(DARK_CLASS) ? 'dark' : 'light'
)

function applyTheme(value: Theme): void {
  if (value === 'dark') {
    document.documentElement.classList.add(DARK_CLASS)
  } else {
    document.documentElement.classList.remove(DARK_CLASS)
  }
}

function setTheme(value: Theme): void {
  theme.value = value
  applyTheme(value)
  try {
    localStorage.setItem(STORAGE_KEY, value)
  } catch {
    // localStorage unavailable; in-memory state still works
  }
}

function toggle(): void {
  setTheme(theme.value === 'dark' ? 'light' : 'dark')
}

// One-time OS preference listener — only acts when user has no explicit choice
const mql = window.matchMedia('(prefers-color-scheme: dark)')
mql.addEventListener('change', (e) => {
  let hasExplicitChoice = false
  try {
    hasExplicitChoice = localStorage.getItem(STORAGE_KEY) !== null
  } catch {
    hasExplicitChoice = false
  }
  if (!hasExplicitChoice) {
    const next: Theme = e.matches ? 'dark' : 'light'
    theme.value = next
    applyTheme(next)
    // Do NOT write to localStorage — keep OS-tracking mode active
  }
})

export function useTheme() {
  return {
    theme: readonly(theme) as Readonly<Ref<Theme>>,
    toggle,
    setTheme,
  }
}
```

Critical implementation notes:
- `const theme = ref<Theme>(...)` MUST be declared at MODULE scope (outside `export function useTheme`) per D-04. This is what makes it a singleton — every caller shares the same Ref.
- The Ref is initialized from `document.documentElement.classList.contains(DARK_CLASS)` so it hydrates from whatever the inline FOUC script (Task 1) already applied.
- The `mql.addEventListener('change', ...)` call is at module scope (not inside an `onMounted` — this is intentional per D-06/D-07; the listener lives for the lifetime of the page, not any single component).
- The OS listener short-circuits when `localStorage.getItem('lexarium:theme') !== null` per D-07.
- `setTheme` performs all three side effects atomically (per D-05): updates Ref, applies class, writes localStorage.
- The storage key `'lexarium:theme'` and class `'my-app-dark'` MUST byte-for-byte match `index.html` (Task 1).
- `readonly(theme) as Readonly<Ref<Theme>>` exposes the Ref as readonly so callers can't mutate `theme.value` directly (they must go through `setTheme`/`toggle`).

Do NOT modify `src/main.ts`. Do NOT modify `src/assets/base.css`. Do NOT create any test file unless explicitly added later.
